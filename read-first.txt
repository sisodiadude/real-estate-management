# Optimization Commands
# This section runs several optimization commands to improve application performance and ensure the app is using the latest compiled files.

# Regenerating Composer autoload files for dependency management.
composer dump-autoload

# Running database migrations to ensure the schema is up to date.
php artisan migrate

# Clearing cached configuration, routes, views, and compiled files.
php artisan config:clear       # Clears cached configuration files
php artisan route:clear        # Clears cached routes
php artisan view:clear         # Clears compiled views
php artisan cache:clear        # Clears application cache
php artisan clear-compiled     # Clears any compiled classes and services

# Optimizing the application by combining and caching configuration and routes.
php artisan optimize

# Caching configuration files for faster loading.
php artisan config:cache

# Seeding the database with necessary data.
php artisan db:seed --class=EmailConfigSeeder
php artisan db:seed --class=AdminSeeder
php artisan db:seed --class=AdminPermissionSeeder
php artisan db:seed --class=AdminHasPermissionSeeder

php artisan config:clear
php artisan route:clear
php artisan view:clear
php artisan cache:clear
php artisan clear-compiled
php artisan optimize
php artisan config:cache
